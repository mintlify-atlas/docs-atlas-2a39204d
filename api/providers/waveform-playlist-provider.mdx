## Overview

The `WaveformPlaylistProvider` is the primary provider for multi-track audio editing. It provides full audio playback, editing capabilities, effects processing, and waveform visualization using Tone.js as the audio engine.

**Use this provider for:**
- Multi-track audio editing
- Complex audio effects chains
- Recording and mixing
- Full playlist manipulation (move, trim, split clips)

**For single-track playback with speed control, use [`MediaElementPlaylistProvider`](/api/providers/media-element-provider) instead.**

## Import

```tsx
import { WaveformPlaylistProvider } from '@waveform-playlist/browser';
```

## Props

<ParamField path="tracks" type="ClipTrack[]" required>
  Array of tracks with clips. Each track contains audio clips positioned on the timeline.
  
  See [ClipTrack type documentation](/api/types/tracks) for details.
</ParamField>

<ParamField path="timescale" type="boolean" default="false">
  Show timescale ruler above waveforms.
</ParamField>

<ParamField path="mono" type="boolean" default="false">
  Render waveforms in mono mode (combine stereo channels).
</ParamField>

<ParamField path="waveHeight" type="number" default="80">
  Height in pixels of each waveform track.
</ParamField>

<ParamField path="samplesPerPixel" type="number" default="1024">
  Initial zoom level. Lower values = more zoomed in. Must match one of the zoom levels.
</ParamField>

<ParamField path="zoomLevels" type="number[]" default="[256, 512, 1024, 2048, 4096, 8192]">
  Array of available zoom levels in samples per pixel. Users can zoom between these levels.
</ParamField>

<ParamField path="automaticScroll" type="boolean" default="false">
  Automatically scroll to keep the playhead centered during playback.
</ParamField>

<ParamField path="theme" type="Partial<WaveformPlaylistTheme>">
  Custom theme to override default colors and styling.
  
  See [Theming documentation](/concepts/theming) for available options.
</ParamField>

<ParamField path="controls" type="{ show: boolean; width: number }" default="{ show: false, width: 0 }">
  Configuration for track control panels (mute, solo, volume, pan).
  
  ```tsx
  controls={{ show: true, width: 150 }}
  ```
</ParamField>

<ParamField path="annotationList" type="object">
  Configuration for annotations display and editing.
  
  <Expandable title="Annotation configuration properties">
    <ResponseField name="annotations" type="AnnotationData[]">
      Array of annotation segments. Must be pre-parsed with numeric `start` and `end` values.
      
      Use `parseAeneas()` from `@waveform-playlist/annotations` to parse raw annotation data.
    </ResponseField>
    
    <ResponseField name="editable" type="boolean" default="false">
      Allow dragging annotation boundaries to edit timing.
    </ResponseField>
    
    <ResponseField name="isContinuousPlay" type="boolean" default="false">
      When enabled, playback continues across annotation boundaries. When disabled, playback stops at the end of the current annotation.
    </ResponseField>
    
    <ResponseField name="linkEndpoints" type="boolean" default="false">
      Link adjacent annotation endpoints. When editing one annotation's end, the next annotation's start moves to match.
    </ResponseField>
    
    <ResponseField name="controls" type="AnnotationAction[]">
      Array of available annotation controls (e.g., delete, merge, split).
    </ResponseField>
  </Expandable>
</ParamField>

<ParamField path="effects" type="EffectsFunction">
  Master effects chain function. Processes all tracks through the specified effects.
  
  See [Audio Effects guide](/guides/audio-effects) for usage.
  
  ```tsx
  effects={(input, output) => {
    const reverb = new Reverb();
    input.connect(reverb).connect(output);
  }}
  ```
</ParamField>

<ParamField path="onReady" type="() => void">
  Callback fired when all tracks have loaded and the playlist is ready for playback.
</ParamField>

<ParamField path="onAnnotationsChange" type="(annotations: AnnotationData[]) => void">
  Callback fired when annotations are edited (drag, split, merge, delete).
  
  **Required** if `annotationList.editable` is `true`. Without this callback, edits will not persist.
  
  ```tsx
  onAnnotationsChange={(updated) => setAnnotations(updated)}
  ```
</ParamField>

<ParamField path="onAnnotationUpdate" type="(annotations: AnnotationData[]) => void" deprecated>
  Deprecated. Use `onAnnotationsChange` instead.
</ParamField>

<ParamField path="barWidth" type="number" default="1">
  Width in pixels of waveform bars.
</ParamField>

<ParamField path="barGap" type="number" default="0">
  Spacing in pixels between waveform bars.
</ParamField>

<ParamField path="progressBarWidth" type="number" default="barWidth + barGap">
  Width in pixels of progress bars during playback. Defaults to fill gaps between bars.
</ParamField>

<ParamField path="onTracksChange" type="(tracks: ClipTrack[]) => void">
  Callback when engine operations (move, trim, split) modify tracks.
  
  **Important:** You must pass the received `tracks` reference back as the `tracks` prop (i.e., `setState(tracks)`). The provider uses reference identity to detect engine-originated updates and skip expensive rebuilds.
  
  ```tsx
  const [tracks, setTracks] = useState(initialTracks);
  
  <WaveformPlaylistProvider
    tracks={tracks}
    onTracksChange={setTracks}  // Pass reference directly
  >
  ```
</ParamField>

## Context Hooks

The provider exposes four separate contexts for optimal performance. Components only re-render when their specific context data changes.

### usePlaybackAnimation

Access playback timing and animation state. This context updates on discrete events (play/pause/stop/seek), not during the animation loop itself.

```tsx
import { usePlaybackAnimation } from '@waveform-playlist/browser';

function MyComponent() {
  const { 
    isPlaying,           // boolean
    currentTime,         // number (seconds)
    currentTimeRef,      // RefObject<number>
    getPlaybackTime      // () => number
  } = usePlaybackAnimation();
}
```

<Expandable title="PlaybackAnimation values">
  <ResponseField name="isPlaying" type="boolean">
    Whether audio is currently playing.
  </ResponseField>
  
  <ResponseField name="currentTime" type="number">
    Current playback position in seconds. Updated on discrete events (play/pause/seek).
  </ResponseField>
  
  <ResponseField name="currentTimeRef" type="RefObject<number>">
    Ref to current time for animation loops. Access via `currentTimeRef.current` in `requestAnimationFrame` callbacks.
  </ResponseField>
  
  <ResponseField name="playbackStartTimeRef" type="RefObject<number>">
    Audio context time when playback started. Used for calculating elapsed time.
  </ResponseField>
  
  <ResponseField name="audioStartPositionRef" type="RefObject<number>">
    Audio position when playback started. Used with elapsed time to calculate current position.
  </ResponseField>
  
  <ResponseField name="getPlaybackTime" type="() => number">
    Returns current playback time from the engine. Auto-wraps at loop boundaries.
    
    Use this in animation loops for accurate timing instead of reading state.
  </ResponseField>
</Expandable>

### usePlaylistState

Access playlist state (selection, loop, annotations).

```tsx
import { usePlaylistState } from '@waveform-playlist/browser';

function MyComponent() {
  const {
    continuousPlay,
    linkEndpoints,
    annotationsEditable,
    isAutomaticScroll,
    isLoopEnabled,
    annotations,
    activeAnnotationId,
    selectionStart,
    selectionEnd,
    selectedTrackId,
    loopStart,
    loopEnd
  } = usePlaylistState();
}
```

<Expandable title="PlaylistState values">
  <ResponseField name="continuousPlay" type="boolean">

  </ResponseField>
  
  <ResponseField name="linkEndpoints" type="boolean">
    Whether adjacent annotation endpoints are linked.
  </ResponseField>
  
  <ResponseField name="annotationsEditable" type="boolean">
    Whether annotations can be edited by dragging.
  </ResponseField>
  
  <ResponseField name="isAutomaticScroll" type="boolean">
    Whether automatic scroll is enabled to keep playhead centered.
  </ResponseField>
  
  <ResponseField name="isLoopEnabled" type="boolean">
    Whether looping is enabled.
  </ResponseField>
  
  <ResponseField name="annotations" type="AnnotationData[]">
    Current annotations array.
  </ResponseField>
  
  <ResponseField name="activeAnnotationId" type="string | null">
    ID of the currently active annotation during playback.
  </ResponseField>
  
  <ResponseField name="selectionStart" type="number">
    Start time of current selection in seconds.
  </ResponseField>
  
  <ResponseField name="selectionEnd" type="number">
    End time of current selection in seconds.
  </ResponseField>
  
  <ResponseField name="selectedTrackId" type="string | null">

### usePlaylistControls

Access control functions for playback, tracks, selection, zoom, and more.

```tsx
import { usePlaylistControls } from '@waveform-playlist/browser';

function MyComponent() {
  const { play, pause, stop, seekTo, setSelection } = usePlaylistControls();
  
  const handlePlay = () => {
    play(); // Play from current position
    // or
    play(5.0); // Play from 5 seconds
    // or
    play(5.0, 10.0); // Play 10 seconds starting at 5s
  };
}
```

### usePlaylistData

Access playlist data (duration, audio buffers, waveform peaks, track states).

```tsx
import { usePlaylistData } from '@waveform-playlist/browser';

function MyComponent() {
  const {
    duration,
    tracks,
    trackStates,
    audioBuffers,
    peaksDataArray,
    sampleRate,
    isReady
  } = usePlaylistData();
}
```

## Complete Example

```tsx
import { useState } from 'react';
import { 
  WaveformPlaylistProvider,
  usePlaylistControls,
  usePlaylistState,
  usePlaylistData
} from '@waveform-playlist/browser';
import { parseAeneas } from '@waveform-playlist/annotations';

// Load your audio and annotations
const initialTracks = [
  {
    id: 'track-1',
    name: 'Vocals',
    clips: [
      {
        id: 'clip-1',
        audioBuffer: vocalsBuffer,  // AudioBuffer from fetch + decodeAudioData
        startSample: 0,
        offsetSamples: 0,
        durationSamples: vocalsBuffer.length,
        sampleRate: vocalsBuffer.sampleRate,
      }
    ],
    volume: 1.0,
    muted: false,
    soloed: false,
    pan: 0,
  },
  {
    id: 'track-2',
    name: 'Instrumental',
    clips: [
      {
        id: 'clip-2',
        audioBuffer: instrumentalBuffer,
        startSample: 0,
        offsetSamples: 0,
        durationSamples: instrumentalBuffer.length,
        sampleRate: instrumentalBuffer.sampleRate,
      }
    ],
    volume: 0.8,
    muted: false,
    soloed: false,
    pan: 0,
  }
];

function PlaybackControls() {
  const { play, pause, stop, seekTo } = usePlaylistControls();
  const { isPlaying, currentTime } = usePlaybackAnimation();
  
  return (
    <div>
      <button onClick={() => play()}>
        {isPlaying ? 'Pause' : 'Play'}
      </button>
      <button onClick={stop}>Stop</button>
      <div>Time: {currentTime.toFixed(2)}s</div>
    </div>
  );
}

function App() {
  const [tracks, setTracks] = useState(initialTracks);
  const [annotations, setAnnotations] = useState([
    parseAeneas({ begin: "0.000", end: "2.500", id: "1", lines: ["First segment"] }),
    parseAeneas({ begin: "2.500", end: "5.000", id: "2", lines: ["Second segment"] })
  ]);
  
  return (
    <WaveformPlaylistProvider
      tracks={tracks}
      onTracksChange={setTracks}
      waveHeight={100}
      samplesPerPixel={1024}
      zoomLevels={[256, 512, 1024, 2048, 4096]}
      automaticScroll
      timescale
      controls={{ show: true, width: 150 }}
      annotationList={{
        annotations,
        editable: true,
        isContinuousPlay: false,
        linkEndpoints: true
      }}
      onAnnotationsChange={setAnnotations}
      onReady={() => console.log('Playlist ready!')}
    >
      <PlaybackControls />
      {/* Your waveform visualization components */}
    </WaveformPlaylistProvider>
  );
}
```
